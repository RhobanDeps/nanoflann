00109                 {

00466         class PooledAllocator
00467         {
00468                 /* We maintain memory alignment to word boundaries by requiring that all
00469                     allocations be in multiples of the machine wordsize.  */
00470                 /* Size of machine word in bytes.  Must be power of 2. */
00471                 /* Minimum number of bytes requested at a time from     the system.  Must be multiple of WORDSIZE. */
00472 
00473 
00474                 size_t  remaining;  /* Number of bytes left in current block of storage. */
00475                 void*   base;     /* Pointer to base of current block of storage. */
00476                 void*   loc;      /* Current location in block to next allocate memory. */
00477 
00478                 void internal_init()
00479                 {
00480                         remaining = 0;
00481                         base = NULL;
00482                         usedMemory = 0;
00483                         wastedMemory = 0;
00484                 }
00485 
00486         public:
00487                 size_t  usedMemory;
00488                 size_t  wastedMemory;
00489 
00493                 PooledAllocator() {
00494                         internal_init();
00495                 }
00496 
00500                 ~PooledAllocator() {
00501                         free_all();
00502                 }
00503 
00505                 void free_all()
00506                 {
00507                         while (base != NULL) {
00508                                 void *prev = *(static_cast<void**>( base)); /* Get pointer to prev block. */
00509                                 ::free(base);
00510                                 base = prev;
00511                         }
00512                         internal_init();
00513                 }
00514 
00519                 void* malloc(const size_t req_size)
00520                 {
00521                         /* Round size up to a multiple of wordsize.  The following expression
00522                             only works for WORDSIZE that is a power of 2, by masking last bits of
00523                             incremented size to zero.
00524                          */
00525                         const size_t size = (req_size + (WORDSIZE - 1)) & ~(WORDSIZE - 1);
00526 
00527                         /* Check whether a new block must be allocated.  Note that the first word
00528                             of a block is reserved for a pointer to the previous block.
00529                          */
00530                         if (size > remaining) {
00531 
00532                                 wastedMemory += remaining;
00533 
00534                                 /* Allocate new storage. */
00535                                 const size_t blocksize = (size + sizeof(void*) + (WORDSIZE-1) > BLOCKSIZE) ?
00536                                                         size + sizeof(void*) + (WORDSIZE-1) : BLOCKSIZE;
00537 
00538                                 // use the standard C malloc to allocate memory
00539                                 void* m = ::malloc(blocksize);
00540                                 if (!m) {
00541                                         fprintf(stderr,"Failed to allocate memory.\n");
00542                                         return NULL;
00543                                 }
00544 
00545                                 /* Fill first word of new block with pointer to previous block. */
00546                                 static_cast<void**>(m)[0] = base;
00547                                 base = m;
00548 
00549                                 size_t shift = 0;
00550                                 //int size_t = (WORDSIZE - ( (((size_t)m) + sizeof(void*)) & (WORDSIZE-1))) & (WORDSIZE-1);
00551 
00552                                 remaining = blocksize - sizeof(void*) - shift;
00553                                 loc = (static_cast<char*>(m) + sizeof(void*) + shift);
00554                         }
00555                         void* rloc = loc;
00556                         loc = static_cast<char*>(loc) + size;
00557                         remaining -= size;
00558 
00559                         usedMemory += size;
00560 
00561                         return rloc;
00562                 }

01207                                                  distance_vector_t& dists, const float epsError) const

01236                         }
01237                         else {
01238                                 bestChild = node->child2;
01239                                 otherChild = node->child1;
01240                                 cut_dist = distance.accum_dist( val, node->sub.divlow, idx);
01241                         }
01242

01244                         searchLevel(result_set, vec, bestChild, mindistsq, dists, epsError);
01245
